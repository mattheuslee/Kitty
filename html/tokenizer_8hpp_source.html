470  void process_math_tokens(std::vector<Token> & tokens) {
471  // Compresses '<' + '=' into '<=' and '>' + '=' into '>='
472  std::vector<Token>::iterator iter = tokens.begin();
473  while (iter != tokens.end()) {
474  std::vector<Token>::iterator next = iter;
475  ++next;
476  if (next != tokens.end()) {
477  if (iter->type == TokenType::LESS) {
478  if (next->type == TokenType::EQUALS) {
479  iter->type = TokenType::L_EQUALS;
480  next = tokens.erase(next);
481  }
482  }
483  else if (iter->type == TokenType::GREATER) {
484  if (next->type == TokenType::EQUALS) {
485  iter->type = TokenType::G_EQUALS;
486  next = tokens.erase(next);
487  }
488  }
489  }
490  iter = next;
491  }
492 
493  // Detects unary '-' tokens
494  for (std::vector<Token>::iterator iter = tokens.begin(); iter != tokens.end(); ++iter) {
495  if (iter->type == TokenType::MATH_SUB) {
496  std::vector<Token>::iterator prev = iter;
497  --prev;
498  if (iter == tokens.begin() ||
499  prev->is_operator() ||
500  prev->is_left_bracket()) {
501  iter->type = TokenType::UNARY_NEG;
502  }
503  }
504  }
505  }

512  Token get_next_token() {
513  // No more tokens
514  if (tokenStartIdx_ == command_.size()) {
515  return Token(TokenType::CMD_END);
516  }
517  // Next token is command word
518  if (isupper(command_[tokenStartIdx_])) {
519  return get_next_command_token();
520  }
521  // Next token is name
522  else if (islower(command_[tokenStartIdx_]) || command_[tokenStartIdx_] == '_') {
523  return get_next_name_token();
524  }
525  // Next token is number
526  else if (isdigit(command_[tokenStartIdx_])) {
527  return get_next_number_token();
528  }
529  // Next token is punctuation
530  else if (validPunctuation_.find(command_[tokenStartIdx_]) != std::string::npos) {
531  return get_next_punctuation_token();
532  }
533  return Token(TokenType::UNKNOWN_TYPE);
534  }

658  void add_missing_optional_arguments() {
659  for (int i = 0; i < commandLookup_.size(); ++i) {
660  int idx = command_.find(commandLookup_[i]);
661  // This command word isn't in the command
662  if (idx == std::string::npos) {
663  continue;
664  }
665  int opParenIdx = command_.find('(', idx + commandLookup_[i].size());
666  int clParenIdx = command_.find(')', opParenIdx);
667  int numArguments = 0;
668  // At least one argument between them
669  if (clParenIdx > opParenIdx + 1) {
670  numArguments = std::count(command_.begin() + opParenIdx, command_.begin() + clParenIdx, ',') + 1;
671  }
672  // Need to fill up arguments
673  if (numArguments < Token(command_str_to_token_type(commandLookup_[i])).num_function_arguments()) {
674  std::string aditionalArguments = get_additional_arguments(command_str_to_token_type(commandLookup_[i]), numArguments);
675  command_.insert(clParenIdx, aditionalArguments);
676  }
677  }
