305  void cleanup_math_tokens(std::vector<Token> & tokens) {
306  // Compresses '<' + '=' into '<=' and '>' + '=' into '>='
307  std::vector<Token>::iterator iter = tokens.begin();
308  while (iter != tokens.end()) {
309  std::vector<Token>::iterator next = iter;
310  ++next;
311  if (next != tokens.end()) {
312  if (iter->type == TokenType::LESS) {
313  if (next->type == TokenType::EQUALS) {
314  iter->type = TokenType::L_EQUALS;
315  next = tokens.erase(next);
316  }
317  }
318  else if (iter->type == TokenType::GREATER) {
319  if (next->type == TokenType::EQUALS) {
320  iter->type = TokenType::G_EQUALS;
321  next = tokens.erase(next);
322  }
323  }
324  }
325  iter = next;
326  }
327 
328  // Detects unary '-' tokens
329  for (std::vector<Token>::iterator iter = tokens.begin(); iter != tokens.end(); ++iter) {
330  if (iter->type == TokenType::MATH_SUB) {
331  std::vector<Token>::iterator prev = iter;
332  --prev;
333  if (iter == tokens.begin() ||
334  is_operator(*prev) ||
335  is_left_bracket(*prev)) {
336  iter->type = TokenType::UNARY_NEG;
337  }
338  }
339  }

342  Token get_next_token() {
343  // No more tokens
344  if (tokenStartIdx_ == command_.size()) {
345  return Token(TokenType::CMD_END);
346  }
347  // Next token is command word
348  if (isupper(command_[tokenStartIdx_])) {
349  return get_next_command_token();
350  }
351  // Next token is name
352  else if (islower(command_[tokenStartIdx_]) || command_[tokenStartIdx_] == '_') {
353  return get_next_name_token();
354  }
355  // Next token is number
356  else if (isdigit(command_[tokenStartIdx_])) {
357  return get_next_number_token();
358  }
359  // Next token is punctuation
360  else if (validPunctuation_.find(command_[tokenStartIdx_]) != std::string::npos) {
361  return get_next_punctuation_token();
362  }
363  return Token(TokenType::UNKNOWN);
364  }

450  void add_missing_optional_arguments() {
451  for (int i = 0; i < commandLookup_.size(); ++i) {
452  int idx = command_.find(commandLookup_[i]);
453  // This command word isn't in the command
454  if (idx == std::string::npos) {
455  continue;
456  }
457  int opParenIdx = command_.find('(', idx + commandLookup_[i].size());
458  int clParenIdx = command_.find(')', opParenIdx);
459  int numArguments = 0;
460  // At least one argument between them
461  if (clParenIdx > opParenIdx + 1) {
462  numArguments = std::count(command_.begin() + opParenIdx, command_.begin() + clParenIdx, ',') + 1;
463  }
464  // Need to fill up arguments
465  if (numArguments < num_function_arguments(command_str_to_token_type(commandLookup_[i]))) {
466  std::string aditionalArguments = get_additional_arguments(command_str_to_token_type(commandLookup_[i]), numArguments);
467  command_.insert(clParenIdx, aditionalArguments);
468  }
469  }
