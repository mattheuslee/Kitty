640  void process_math_tokens(std::vector<Token> & tokens) {
641  // Compresses '<' + '=' into '<=' and '>' + '=' into '>='
642  std::vector<Token>::iterator iter = tokens.begin();
643  while (iter != tokens.end()) {
644  std::vector<Token>::iterator next = iter;
645  ++next;
646  if (next != tokens.end()) {
647  if (iter->type == TokenType::LESS) {
648  if (next->type == TokenType::EQUALS) {
649  iter->type = TokenType::L_EQUALS;
650  next = tokens.erase(next);
651  }
652  }
653  else if (iter->type == TokenType::GREATER) {
654  if (next->type == TokenType::EQUALS) {
655  iter->type = TokenType::G_EQUALS;
656  next = tokens.erase(next);
657  }
658  }
659  }
660  iter = next;
661  }
662 
663  // Detects unary '-' tokens
664  for (std::vector<Token>::iterator iter = tokens.begin(); iter != tokens.end(); ++iter) {
665  if (iter->type == TokenType::MATH_SUB) {
666  std::vector<Token>::iterator prev = iter;
667  --prev;
668  if (iter == tokens.begin() ||
669  prev->is_operator() ||
670  prev->is_left_bracket()) {
671  iter->type = TokenType::UNARY_NEG;
672  }
673  }
674  }
675  }

682  Token get_next_token() {
683  // No more tokens
684  if (tokenStartIdx_ == command_.size()) {
685  return Token(TokenType::CMD_END);
686  }
687  // Next token is command word
688  if (isupper(command_[tokenStartIdx_])) {
689  return get_next_command_token();
690  }
691  // Next token is name
692  else if (islower(command_[tokenStartIdx_]) || command_[tokenStartIdx_] == '_') {
693  return get_next_name_token();
694  }
695  // Next token is number
696  else if (isdigit(command_[tokenStartIdx_])) {
697  return get_next_number_token();
698  }
699  // Next token is string
700  else if (command_[tokenStartIdx_] == '"') {
701  return get_next_string_token('"');
702  }
703  else if (command_[tokenStartIdx_] == '\'') {
704  return get_next_string_token('\'');
705  }
706  // Next token is punctuation
707  else if (validPunctuation_.find(command_[tokenStartIdx_]) != std::string::npos) {
708  return get_next_punctuation_token();
709  }
710  return Token(TokenType::UNKNOWN_TYPE);
711  }
712 
719  Token get_next_command_token() {
720  for (int i = 0; i < commandLookup_.size(); ++i) {
721  if (command_.find(commandLookup_[i], tokenStartIdx_) == tokenStartIdx_) {
722  tokenStartIdx_ += commandLookup_[i].size();
723  return Token(command_str_to_token_type(commandLookup_[i]));
724  }
725  }
726  return Token(TokenType::UNKNOWN_TYPE);
727  }

735  Token get_next_name_token() {
736  int currIdx = tokenStartIdx_;
737  while (currIdx < command_.size() && (islower(command_[currIdx]) || command_[currIdx] == '_')) {
738  ++currIdx;
739  }
740  std::string value = command_.substr(tokenStartIdx_, currIdx - tokenStartIdx_);
741  Token result(TokenType::NAME, value);

853  void add_missing_optional_arguments() {
854  for (int i = 0; i < commandLookup_.size(); ++i) {
855  int idx = command_.find(commandLookup_[i]);
856  // This command word isn't in the command
857  if (idx == std::string::npos) {
858  continue;
859  }
860  int opParenIdx = command_.find('(', idx + commandLookup_[i].size());
861  int clParenIdx = command_.find(')', opParenIdx);
862  int numArguments = 0;
863  // At least one argument between them
864  if (clParenIdx > opParenIdx + 1) {
865  numArguments = std::count(command_.begin() + opParenIdx, command_.begin() + clParenIdx, ',') + 1;
866  }
867  // Need to fill up arguments
868  if (numArguments < Token(command_str_to_token_type(commandLookup_[i])).num_function_arguments()) {
869  std::string aditionalArguments = get_additional_arguments(command_str_to_token_type(commandLookup_[i]), numArguments);
870  command_.insert(clParenIdx, aditionalArguments);
871  }
872  }
873  }
