118  void process_math_tokens(Deque<Token> & tokens) {
119  Log.verbose(F("%s\n"), PRINT_FUNC);
120  // Compresses '<' + '=' into '<=' and '>' + '=' into '>='
121  typename Deque<Token>::Iterator iter = tokens.begin();
122  while (iter != tokens.end()) {
123  typename Deque<Token>::Iterator next = iter;
124  ++next;
125  if (next != tokens.end()) {
126  if (iter->is_less() && next->is_equals()) {
127  Log.verbose(F("%s: compressing into <=\n"), PRINT_FUNC);
128  iter->set_type(TokenType::L_EQUALS);
129  next = tokens.erase(next);
130  }
131  else if (iter->is_greater() && next->is_equals()) {
132  Log.verbose(F("%s: compressing into >=\n"), PRINT_FUNC);
133  iter->set_type(TokenType::G_EQUALS);
134  next = tokens.erase(next);
135  }
136  }
137  iter = next;
138  }
139 
140  // Detects unary '-' tokens
141  for (iter = tokens.begin(); iter != tokens.end(); ++iter) {
142  if (iter->is_math_sub()) {
143  typename Deque<Token>::Iterator prev = iter;
144  --prev;
145  if (iter == tokens.begin() ||
146  prev->is_operator() ||
147  prev->is_op_paren()) {
148  Log.verbose(F("%s: changing to unary -\n"), PRINT_FUNC);
149  iter->set_type(TokenType::UNARY_NEG);
150  }
151  }
152  }
153  }

160  Token get_next_token() {
161  Log.verbose(F("%s\n"), PRINT_FUNC);
162  // No more tokens
163  if (tokenStartIdx_ >= command_.strlen()) {
164  return Token(TokenType::CMD_END, *getPoolFunc_);
165  }
166  // Next token is command word
167  if (isupper(command_[tokenStartIdx_])) {
168  return get_next_command_token();
169  }
170  // Next token is name
171  else if (islower(command_[tokenStartIdx_]) || command_[tokenStartIdx_] == '_') {
172  return get_next_name_token();
173  }
174  // Next token is number
175  else if (isdigit(command_[tokenStartIdx_])) {
176  return get_next_number_token();
177  }
178  // Next token is string
179  else if (command_[tokenStartIdx_] == '"') {
180  return get_next_string_token('"');
181  }
182  else if (command_[tokenStartIdx_] == '\'') {
183  return get_next_string_token('\'');
184  }
185  // Next token is punctuation
186  else if (validPunctuation_.find(command_[tokenStartIdx_]) != -1) {
187  return get_next_punctuation_token();
188  }
189  ++tokenStartIdx_;
190  Log.warning(F("%s: unknown token %c\n"), PRINT_FUNC, command_[tokenStartIdx_ - 1]);
191  return Token(TokenType::UNKNOWN_TOKEN, *getPoolFunc_);
192  }

200  Token get_next_command_token() {
201  Log.verbose(F("%s\n"), PRINT_FUNC);
202  Token token(TokenType::UNKNOWN_TOKEN, *getPoolFunc_);
203  for (int i = 0; i < commandLookupSize_; ++i) {
204  if (command_.find(get_command_lookup()[i], tokenStartIdx_) == tokenStartIdx_) {
205  tokenStartIdx_ += ::strlen(get_command_lookup()[i]);
206  token.set_type(command_str_to_token_type(get_command_lookup()[i]));
207  return token;
208  }
209  }
210  Log.warning(F("%s: not a valid command\n"), PRINT_FUNC);
211  // Not a valid command,
212  // Skip forward until the end of this word
213  for (++tokenStartIdx_; tokenStartIdx_ < command_.strlen() && islower(command_[tokenStartIdx_]); ++tokenStartIdx_);
214  return token;
215  }

323  void add_missing_optional_arguments() {
324  Log.verbose(F("%s\n"), PRINT_FUNC);
325  for (int i = 0; i < commandLookupSize_; ++i) {
326  int idx = command_.find(get_command_lookup()[i]);
327  if (idx == -1) {
328  continue;
329  }
330  int opParenIdx = command_.find("(", idx + ::strlen(get_command_lookup()[i]));
331  int clParenIdx = command_.find(')', opParenIdx);
332  int numArguments = 0;
333  // At least one argument between them
334  if (clParenIdx > opParenIdx + 1) {
335  numArguments = command_.count(',', opParenIdx, clParenIdx) + 1;
336  }
337  // Need to fill up arguments
338  TokenType tokenType = command_str_to_token_type(get_command_lookup()[i]);
339  int requiredArguments = Token(tokenType, *getPoolFunc_).num_function_arguments();
340  if (numArguments < requiredArguments) {
341  PoolString additionalArguments(get_additional_arguments(tokenType, numArguments));
342  command_.insert(additionalArguments.c_str(), clParenIdx);
343  }
344  }
