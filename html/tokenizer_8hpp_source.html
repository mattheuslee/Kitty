593  void process_math_tokens(std::vector<Token> & tokens) {
594  // Compresses '<' + '=' into '<=' and '>' + '=' into '>='
595  std::vector<Token>::iterator iter = tokens.begin();
596  while (iter != tokens.end()) {
597  std::vector<Token>::iterator next = iter;
598  ++next;
599  if (next != tokens.end()) {
600  if (iter->type == TokenType::LESS) {
601  if (next->type == TokenType::EQUALS) {
602  iter->type = TokenType::L_EQUALS;
603  next = tokens.erase(next);
604  }
605  }
606  else if (iter->type == TokenType::GREATER) {
607  if (next->type == TokenType::EQUALS) {
608  iter->type = TokenType::G_EQUALS;
609  next = tokens.erase(next);
610  }
611  }
612  }
613  iter = next;
614  }
615 
616  // Detects unary '-' tokens
617  for (std::vector<Token>::iterator iter = tokens.begin(); iter != tokens.end(); ++iter) {
618  if (iter->type == TokenType::MATH_SUB) {
619  std::vector<Token>::iterator prev = iter;
620  --prev;
621  if (iter == tokens.begin() ||
622  prev->is_operator() ||
623  prev->is_left_bracket()) {
624  iter->type = TokenType::UNARY_NEG;
625  }
626  }
627  }
628  }

635  Token get_next_token() {
636  // No more tokens
637  if (tokenStartIdx_ == command_.size()) {
638  return Token(TokenType::CMD_END);
639  }
640  // Next token is command word
641  if (isupper(command_[tokenStartIdx_])) {
642  return get_next_command_token();
643  }
644  // Next token is name
645  else if (islower(command_[tokenStartIdx_]) || command_[tokenStartIdx_] == '_') {
646  return get_next_name_token();
647  }
648  // Next token is number
649  else if (isdigit(command_[tokenStartIdx_])) {
650  return get_next_number_token();
651  }
652  // Next token is string
653  else if (command_[tokenStartIdx_] == '"') {
654  return get_next_string_token('"');
655  }
656  else if (command_[tokenStartIdx_] == '\'') {
657  return get_next_string_token('\'');
658  }
659  // Next token is punctuation
660  else if (validPunctuation_.find(command_[tokenStartIdx_]) != std::string::npos) {
661  return get_next_punctuation_token();
662  }
663  return Token(TokenType::UNKNOWN_TYPE);
664  }
665 
672  Token get_next_command_token() {
673  for (int i = 0; i < commandLookup_.size(); ++i) {
674  if (command_.find(commandLookup_[i], tokenStartIdx_) == tokenStartIdx_) {
675  tokenStartIdx_ += commandLookup_[i].size();
676  return Token(command_str_to_token_type(commandLookup_[i]));
677  }
678  }
679  return Token(TokenType::UNKNOWN_TYPE);
680  }

806  void add_missing_optional_arguments() {
807  for (int i = 0; i < commandLookup_.size(); ++i) {
808  int idx = command_.find(commandLookup_[i]);
809  // This command word isn't in the command
810  if (idx == std::string::npos) {
811  continue;
812  }
813  int opParenIdx = command_.find('(', idx + commandLookup_[i].size());
814  int clParenIdx = command_.find(')', opParenIdx);
815  int numArguments = 0;
816  // At least one argument between them
817  if (clParenIdx > opParenIdx + 1) {
818  numArguments = std::count(command_.begin() + opParenIdx, command_.begin() + clParenIdx, ',') + 1;
819  }
820  // Need to fill up arguments
821  if (numArguments < Token(command_str_to_token_type(commandLookup_[i])).num_function_arguments()) {
822  std::string aditionalArguments = get_additional_arguments(command_str_to_token_type(commandLookup_[i]), numArguments);
823  command_.insert(clParenIdx, aditionalArguments);
824  }
825  }
