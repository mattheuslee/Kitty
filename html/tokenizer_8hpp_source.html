501  void process_math_tokens(std::vector<Token> & tokens) {
502  // Compresses '<' + '=' into '<=' and '>' + '=' into '>='
503  std::vector<Token>::iterator iter = tokens.begin();
504  while (iter != tokens.end()) {
505  std::vector<Token>::iterator next = iter;
506  ++next;
507  if (next != tokens.end()) {
508  if (iter->type == TokenType::LESS) {
509  if (next->type == TokenType::EQUALS) {
510  iter->type = TokenType::L_EQUALS;
511  next = tokens.erase(next);
512  }
513  }
514  else if (iter->type == TokenType::GREATER) {
515  if (next->type == TokenType::EQUALS) {
516  iter->type = TokenType::G_EQUALS;
517  next = tokens.erase(next);
518  }
519  }
520  }
521  iter = next;
522  }
523 
524  // Detects unary '-' tokens
525  for (std::vector<Token>::iterator iter = tokens.begin(); iter != tokens.end(); ++iter) {
526  if (iter->type == TokenType::MATH_SUB) {
527  std::vector<Token>::iterator prev = iter;
528  --prev;
529  if (iter == tokens.begin() ||
530  prev->is_operator() ||
531  prev->is_left_bracket()) {
532  iter->type = TokenType::UNARY_NEG;
533  }
534  }
535  }
536  }

543  Token get_next_token() {
544  // No more tokens
545  if (tokenStartIdx_ == command_.size()) {
546  return Token(TokenType::CMD_END);
547  }
548  // Next token is command word
549  if (isupper(command_[tokenStartIdx_])) {
550  return get_next_command_token();
551  }
552  // Next token is name
553  else if (islower(command_[tokenStartIdx_]) || command_[tokenStartIdx_] == '_') {
554  return get_next_name_token();
555  }
556  // Next token is number
557  else if (isdigit(command_[tokenStartIdx_])) {
558  return get_next_number_token();
559  }
560  // Next token is punctuation
561  else if (validPunctuation_.find(command_[tokenStartIdx_]) != std::string::npos) {
562  return get_next_punctuation_token();
563  }
564  return Token(TokenType::UNKNOWN_TYPE);
565  }

689  void add_missing_optional_arguments() {
690  for (int i = 0; i < commandLookup_.size(); ++i) {
691  int idx = command_.find(commandLookup_[i]);
692  // This command word isn't in the command
693  if (idx == std::string::npos) {
694  continue;
695  }
696  int opParenIdx = command_.find('(', idx + commandLookup_[i].size());
697  int clParenIdx = command_.find(')', opParenIdx);
698  int numArguments = 0;
699  // At least one argument between them
700  if (clParenIdx > opParenIdx + 1) {
701  numArguments = std::count(command_.begin() + opParenIdx, command_.begin() + clParenIdx, ',') + 1;
702  }
703  // Need to fill up arguments
704  if (numArguments < Token(command_str_to_token_type(commandLookup_[i])).num_function_arguments()) {
705  std::string aditionalArguments = get_additional_arguments(command_str_to_token_type(commandLookup_[i]), numArguments);
706  command_.insert(clParenIdx, aditionalArguments);
707  }
708  }
