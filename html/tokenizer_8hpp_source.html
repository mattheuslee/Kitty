537  void process_math_tokens(std::vector<Token> & tokens) {
538  // Compresses '<' + '=' into '<=' and '>' + '=' into '>='
539  std::vector<Token>::iterator iter = tokens.begin();
540  while (iter != tokens.end()) {
541  std::vector<Token>::iterator next = iter;
542  ++next;
543  if (next != tokens.end()) {
544  if (iter->type == TokenType::LESS) {
545  if (next->type == TokenType::EQUALS) {
546  iter->type = TokenType::L_EQUALS;
547  next = tokens.erase(next);
548  }
549  }
550  else if (iter->type == TokenType::GREATER) {
551  if (next->type == TokenType::EQUALS) {
552  iter->type = TokenType::G_EQUALS;
553  next = tokens.erase(next);
554  }
555  }
556  }
557  iter = next;
558  }
559 
560  // Detects unary '-' tokens
561  for (std::vector<Token>::iterator iter = tokens.begin(); iter != tokens.end(); ++iter) {
562  if (iter->type == TokenType::MATH_SUB) {
563  std::vector<Token>::iterator prev = iter;
564  --prev;
565  if (iter == tokens.begin() ||
566  prev->is_operator() ||
567  prev->is_left_bracket()) {
568  iter->type = TokenType::UNARY_NEG;
569  }
570  }
571  }
572  }

579  Token get_next_token() {
580  // No more tokens
581  if (tokenStartIdx_ == command_.size()) {
582  return Token(TokenType::CMD_END);
583  }
584  // Next token is command word
585  if (isupper(command_[tokenStartIdx_])) {
586  return get_next_command_token();
587  }
588  // Next token is name
589  else if (islower(command_[tokenStartIdx_]) || command_[tokenStartIdx_] == '_') {
590  return get_next_name_token();
591  }
592  // Next token is number
593  else if (isdigit(command_[tokenStartIdx_])) {
594  return get_next_number_token();
595  }
596  // Next token is punctuation
597  else if (validPunctuation_.find(command_[tokenStartIdx_]) != std::string::npos) {
598  return get_next_punctuation_token();
599  }
600  return Token(TokenType::UNKNOWN_TYPE);
601  }

609  Token get_next_command_token() {
610  for (int i = 0; i < commandLookup_.size(); ++i) {
611  if (command_.find(commandLookup_[i], tokenStartIdx_) == tokenStartIdx_) {
612  tokenStartIdx_ += commandLookup_[i].size();

615  }

725  void add_missing_optional_arguments() {
726  for (int i = 0; i < commandLookup_.size(); ++i) {
727  int idx = command_.find(commandLookup_[i]);
728  // This command word isn't in the command
729  if (idx == std::string::npos) {
730  continue;
731  }
732  int opParenIdx = command_.find('(', idx + commandLookup_[i].size());
733  int clParenIdx = command_.find(')', opParenIdx);
734  int numArguments = 0;
735  // At least one argument between them
736  if (clParenIdx > opParenIdx + 1) {
737  numArguments = std::count(command_.begin() + opParenIdx, command_.begin() + clParenIdx, ',') + 1;
738  }
739  // Need to fill up arguments
740  if (numArguments < Token(command_str_to_token_type(commandLookup_[i])).num_function_arguments()) {
741  std::string aditionalArguments = get_additional_arguments(command_str_to_token_type(commandLookup_[i]), numArguments);
742  command_.insert(clParenIdx, aditionalArguments);
743  }
744  }
