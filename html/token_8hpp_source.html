783  "",
784  "",

786  "If",
787  "Else",
788  };
789  static const int numTypes = sizeof(lookup) / sizeof(lookup[0]);
790  for (int i = 0; i < numTypes; ++i) {
791  if (std::strcmp(str, lookup[i]) == 0) {
792  Log.verbose(F("%s: %d\n"), PRINT_FUNC, static_cast<TokenType>(i));

794  }
795  }

797  Log.warning(F("%s: unknown command string %s\n"), PRINT_FUNC, str);
