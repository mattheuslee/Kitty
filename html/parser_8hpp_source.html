111  Log.verbose(F("%s\n"), PRINT_FUNC);
112  Deque<Token> operatorStack(*getAllocFunc_);
113  Deque<Token> output(*getAllocFunc_);
114 
115  for (typename Deque<Token>::Iterator it = command_.begin(); it != command_.end(); ++it) {
116  Token & token = *it;
117  if (token.is_operand()) {
118  Log.verbose(F("%s: operand\n"), PRINT_FUNC);
119  Log.verbose(F("%s: operand %s pushed to output\n"), PRINT_FUNC, token.str().c_str());
120  output.push_back(token);
121  }
122  else if (token.is_function()) {
123  Log.verbose(F("%s: function\n"), PRINT_FUNC);
124  Log.verbose(F("%s: function %s pushed to operator stack\n"), PRINT_FUNC, token.str().c_str());
125  operatorStack.push_back(token);
126  }
127  else if (token.is_operator()) {
128  Log.verbose(F("%s: operator\n"), PRINT_FUNC);
129  while (!operatorStack.is_empty() &&
130  (operatorStack.back().is_function() ||
131  operatorStack.back().has_greater_precedence_than(token) ||
132  (operatorStack.back().has_equal_precedence_to(token) &&
133  operatorStack.back().is_left_associative())) &&
134  !operatorStack.back().is_op_paren()) {
135  Log.verbose(F("%s: operator %s pushed from operator stack to output\n"), PRINT_FUNC, operatorStack.back().str().c_str());
136  output.push_back(operatorStack.back());
137  operatorStack.pop_back();
138  }
139  Log.verbose(F("%s: operator %s pushed to operator stack\n"), PRINT_FUNC, token.str().c_str());
140  operatorStack.push_back(token);
141  }
142  else if (token.is_op_paren()) {
143  Log.verbose(F("%s: op_paren\n"), PRINT_FUNC);
144  Log.verbose(F("%s: op paren %s pushed to operator stack\n"), PRINT_FUNC, token.str().c_str());
145  operatorStack.push_back(token);
146  }
147  else if (token.is_cl_paren()) {
148  Log.verbose(F("%s: cl_paren\n"), PRINT_FUNC);
149  while (!operatorStack.is_empty() &&
150  !operatorStack.back().is_op_paren()) {
151  Log.verbose(F("%s: %s pushed from operator stack to output\n"), PRINT_FUNC, operatorStack.back().str().c_str());
152  output.push_back(operatorStack.back());
153  operatorStack.pop_back();
154  }
155  Log.verbose(F("%s: %s popped from operator stack\n"), PRINT_FUNC, operatorStack.back().str().c_str());
156  operatorStack.pop_back();
157  }
158  else if (token.is_comma()) {
159  Log.verbose(F("%s: comma\n"), PRINT_FUNC);
160  while (!operatorStack.is_empty() &&
161  !operatorStack.back().is_op_paren()) {
162  Log.verbose(F("%s: %s pushed from operator stack to output\n"), PRINT_FUNC, operatorStack.back().str().c_str());
163  output.push_back(operatorStack.back());
164  operatorStack.pop_back();
165  }
166  }
167  else {
168  Log.verbose(F("%s: others\n"), PRINT_FUNC);
169  Log.verbose(F("%s: %s pushed to output\n"), PRINT_FUNC, token.str().c_str());
170  output.push_back(token);
171  }
172  }
173  while (!operatorStack.is_empty()) {
174  Log.verbose(F("%s: %s pushed from operator stack to output\n"), PRINT_FUNC, operatorStack.back().str().c_str());
175  output.push_back(operatorStack.back());
176  operatorStack.pop_back();
177  }
