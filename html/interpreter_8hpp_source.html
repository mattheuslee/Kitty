461  std::stack<Token> evaluate_postfix(std::queue<Token> & tokenQueue) {
462  std::stack<Token> tokenStack;
463 
464  while (!tokenQueue.empty()) {
465  Token token = tokenQueue.front();
466  tokenQueue.pop();
467  if (token.is_unary_operator()) {
468  Token operand = tokenStack.top();
469  tokenStack.pop();
470  Token result = evaluate_unary_operation(token, operand);
471  tokenStack.push(result);
472  }
473  else if (token.is_operator()) {
474  Token rhs = tokenStack.top();
475  tokenStack.pop();
476  Token lhs = tokenStack.top();
477  tokenStack.pop();
478  Token result = evaluate_operation(token, lhs, rhs);
479  tokenStack.push(result);
480  }
481  else if (token.is_operand()) {
482  tokenStack.push(token);
483  }
484  else if (token.is_function()) {
485  tokenStack.push(token);
486  }
487  }
488  return tokenStack;
489  }

535  Token evaluate_operation(Token const & op, Token const & lhs, Token const & rhs) {
536  int lhsValue = get_token_value(lhs);
537  int rhsValue = get_token_value(rhs);
538  Token result(TokenType::NUM_VAL);
539  result.value = "0";
540 
541  if (op.type == TokenType::EQUALS) {
542  result.value = int_to_str(lhsValue == rhsValue);
543  }
544  else if (op.type == TokenType::L_EQUALS) {
545  result.value = int_to_str(lhsValue <= rhsValue);
546  }
547  else if (op.type == TokenType::G_EQUALS) {
548  result.value = int_to_str(lhsValue >= rhsValue);
549  }
550  else if (op.type == TokenType::LESS) {
551  result.value = int_to_str(lhsValue < rhsValue);
552  }
553  else if (op.type == TokenType::GREATER) {
554  result.value = int_to_str(lhsValue > rhsValue);
555  }
556  else if (op.type == TokenType::MATH_ADD) {
557  result.value = int_to_str(lhsValue + rhsValue);
558  }
559  else if (op.type == TokenType::MATH_SUB) {
560  result.value = int_to_str(lhsValue - rhsValue);
561  }
562  else if (op.type == TokenType::MATH_MUL) {
563  result.value = int_to_str(lhsValue * rhsValue);
564  }
565  else if (op.type == TokenType::MATH_DIV) {
566  result.value = int_to_str(lhsValue / rhsValue);
567  }
568  else if (op.type == TokenType::MATH_MOD) {
569  result.value = int_to_str(lhsValue % rhsValue);
570  }
571  else if (op.type == TokenType::MATH_POW) {
572  result.value = int_to_str(power(lhsValue, rhsValue));
573  }
574  else if (op.type == TokenType::LOGI_AND) {
575  result.value = int_to_str(lhsValue && rhsValue);
576  }
577  else if (op.type == TokenType::LOGI_OR) {
578  result.value = int_to_str(lhsValue || rhsValue);
579  }
580  return result;
581  }
