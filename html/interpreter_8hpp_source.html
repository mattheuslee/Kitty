311  void execute(std::vector<Token> const & command) {
312  // Nothing to execute
313  if (command.empty()) {
314  return;
315  }
316  if (command.back().is_if()) {
317  execute_if(command);
318  }
319  else if (command.back().is_else()) {
320  execute_else(command);
321  }
322  else {
323  // For every other command, last condition at this scope level becomes null
324  lastCondition_[currScopeLevel_] = -1;
325  if (command[0].is_name()) {
326  // Printing information
327  if (command.size() == 1) {
328  execute_print_info(command);
329  }
330  // Creation of a device/group
331  else if (command.back().is_create_num() ||
332  command.back().is_create_led() ||
333  command.back().is_create_servo() ||
334  command.back().is_create_group()) {
335  execute_create(command);
336  }
337  else if (command.back().is_move_by() ||
338  command.back().is_move_by_for()) {
339  execute_move_by(command);
340  }
341  // Running group
342  else {
343  execute_run_group(command);
344  }
345  }
346  else if (command[0].is_string()) {
347  execute_print_string(command);
348  }
349  }

457  void execute_move_by(std::vector<Token> const & command) {
458  std::queue<Token> tokenQueue;
459  // Skip the move by or move by for token at the end
460  for (int i = 0; i < command.size() - 1; ++i) {
461  tokenQueue.push(command[i]);
462  }
463  poolstring_t name(stringPool_, tokenQueue.front().value.c_str());
464  // Nothing to move
465  int deviceIdx = device_exists(name);
466  if (deviceIdx == -1) {
467  Serial.print(F("Error: "));
468  Serial.print(name.c_str());
469  Serial.println(F(" does not exist"));
470  return;
471  }
472  // Cannot be moved
473  DeviceType deviceType = deviceTypes_[deviceIdx];
474  if (deviceType != DeviceType::NUM &&
475  deviceType != DeviceType::LED &&
476  deviceType != DeviceType::SERVO) {
477  Serial.print(F("Error: move command is not valid for"));
478  Serial.println(name.c_str());
479  return;
480  }
481  tokenQueue.pop(); // Pop off move by command
482 
483  std::stack<Token> result = evaluate_postfix(tokenQueue);
484 
485  int displacement, durationMs = 0;
486  // There will be an additional argument if it is move by for
487  if (result.size() > 1) {
488  durationMs = get_token_value(result.top());
489  result.pop();
490  }
491  displacement = get_token_value(result.top());
492 
493  // Execute move
494  int & deviceInfo1 = deviceInfo_1_[deviceIdx];
495  int & deviceInfo2 = deviceInfo_2_[deviceIdx];
496  deviceInfo2 += displacement;
497  switch (deviceType) {
498  case NUM:
499  break;
500  case LED:
501  analogWrite(deviceInfo1, deviceInfo2 * 2.55);
502  break;
503  };
504  if (command.back().is_move_by_for()) {
505  delay(durationMs);
506  deviceInfo2 -= displacement;
507  switch (deviceType) {
508  case NUM:
509  break;
510  case LED:
511  analogWrite(deviceInfo1, deviceInfo2 * 2.55);
512  break;
513  };
514  }
515  }

525  void execute_run_group(std::vector<Token> const & command) {
526  std::queue<Token> tokenQueue;
527  for (int i = 0; i < command.size() - 1; ++i) {
528  tokenQueue.push(command[i]);
529  }
530  // Extract name of group and check if it exists
531  poolstring_t name(stringPool_, tokenQueue.front().value.c_str());
532  int groupIdx = group_exists(name);
533  if (groupIdx == -1) {
534  Log.warning(F("%s: %s does not exist\n"), PRINT_FUNC, name.c_str());
535  return;
536  }
537  // Extract number of times to run group
538  std::stack<Token> result = evaluate_postfix(tokenQueue);
539  int numTimes = get_token_value(result.top());
540  Log.trace(F("%s: Running group %s %d times\n"), PRINT_FUNC, name.c_str(), numTimes);
541  // Push command for one more call to run the group
542  if (numTimes > 1) {
543  commandQueue_.push_front(poolstring_t(stringPool_));
544  commandQueue_.front() += name.c_str();
545  commandQueue_.front() += "RunGroup(";
546  commandQueue_.front() += int_to_str(numTimes - 1, stringPool_);
547  commandQueue_.front() += ")";
548  Log.trace(F("%s: pushed %s to the front of the command queue\n"), PRINT_FUNC, commandQueue_.front().c_str());
549  }
550  // Continuously run the group
551  else if (numTimes == -1) {
552  commandQueue_.push_front(poolstring_t(stringPool_));
553  commandQueue_.front() += name.c_str();
554  commandQueue_.front() += "RunGroup(";
555  commandQueue_.front() += "-1";
556  commandQueue_.front() += ")";
557  Log.trace(F("%s: pushed %s to the front of the command queue\n"), PRINT_FUNC, commandQueue_.front().c_str());
558  }
559  // Could get the instruction to run the group 0 times
560  if (numTimes == -1 || numTimes > 0) {
561  // Push commands from group to command queue
562  int numGroupCommands = groupCommands_.size(groupIdx);
563  Log.trace(F("%s: %d commands to add to command queue\n"), PRINT_FUNC, numGroupCommands);
564  for (int i = numGroupCommands - 1; i >= 0; --i) {
565  int commandStringPoolIdx = groupCommands_.get_str_idx(groupIdx, i);
566  poolstring_t commandString(stringPool_, commandStringPoolIdx);
567  Log.trace(F("%s: pushing %s to front of command queue\n"), PRINT_FUNC, commandString.c_str());
568  commandQueue_.push_front(commandString);
569  }
570  }
571  }

592  std::stack<Token> evaluate_postfix(std::queue<Token> & tokenQueue) {
593  std::stack<Token> tokenStack;
594 
595  while (!tokenQueue.empty()) {
596  Token token = tokenQueue.front();
597  tokenQueue.pop();
598  if (token.is_unary_operator()) {
599  Token operand = tokenStack.top();
600  tokenStack.pop();
601  Token result = evaluate_unary_operation(token, operand);
602  tokenStack.push(result);
603  }
604  else if (token.is_operator()) {
605  Token rhs = tokenStack.top();
606  tokenStack.pop();
607  Token lhs = tokenStack.top();
608  tokenStack.pop();
609  Token result = evaluate_operation(token, lhs, rhs);
610  tokenStack.push(result);
611  }
612  else if (token.is_operand()) {
613  tokenStack.push(token);
614  }
615  else if (token.is_function()) {
616  tokenStack.push(token);
617  }
618  }
619  return tokenStack;
620  }

666  Token evaluate_operation(Token const & op, Token const & lhs, Token const & rhs) {
667  int lhsValue = get_token_value(lhs);
668  int rhsValue = get_token_value(rhs);
669  Token result(TokenType::NUM_VAL);
670  result.value = "0";
671 
672  if (op.type == TokenType::EQUALS) {
673  result.value = int_to_str(lhsValue == rhsValue);
674  }
675  else if (op.type == TokenType::L_EQUALS) {
676  result.value = int_to_str(lhsValue <= rhsValue);
677  }
678  else if (op.type == TokenType::G_EQUALS) {
679  result.value = int_to_str(lhsValue >= rhsValue);
680  }
681  else if (op.type == TokenType::LESS) {
682  result.value = int_to_str(lhsValue < rhsValue);
683  }
684  else if (op.type == TokenType::GREATER) {
685  result.value = int_to_str(lhsValue > rhsValue);
686  }
687  else if (op.type == TokenType::MATH_ADD) {
688  result.value = int_to_str(lhsValue + rhsValue);
689  }
690  else if (op.type == TokenType::MATH_SUB) {
691  result.value = int_to_str(lhsValue - rhsValue);
692  }
693  else if (op.type == TokenType::MATH_MUL) {
694  result.value = int_to_str(lhsValue * rhsValue);
695  }
696  else if (op.type == TokenType::MATH_DIV) {
697  result.value = int_to_str(lhsValue / rhsValue);
698  }
699  else if (op.type == TokenType::MATH_MOD) {
700  result.value = int_to_str(lhsValue % rhsValue);
701  }
702  else if (op.type == TokenType::MATH_POW) {
703  result.value = int_to_str(power(lhsValue, rhsValue));
704  }
705  else if (op.type == TokenType::LOGI_AND) {
706  result.value = int_to_str(lhsValue && rhsValue);
707  }
708  else if (op.type == TokenType::LOGI_OR) {
709  result.value = int_to_str(lhsValue || rhsValue);
710  }
711  return result;
712  }

854  void add_to_if(poolstring_t const & command) {
855  std::vector<Token> tokens = tokenizer_.tokenize(std::string(command.c_str()));
856  // Group is closed
857  if (bracketParity_ == 0 && tokens.size() == 2 && tokens[0].is_right_bracket()) {
858  close_if();
859  }
860  // Add command to group
861  else {
862  commandBuffer_.push_back(command);
863  // Check if command introduces a '('
864  if (tokens[tokens.size() - 2].is_left_bracket()) {
865  ++bracketParity_;
866  }
867  // Check if command closes with a ')'
868  else if (tokens.size() == 2 && tokens[0].is_right_bracket()) {
869  --bracketParity_;
870  }
871  }
872  }

878  void close_if() {
879  int condition = lastCondition_[currScopeLevel_ - 1];
880  if (condition) {
881  Log.trace(F("%s: condition is true\n"), PRINT_FUNC);
882  // Special interpreter-only command to ensure we decrease scope level
883  // after all the instructions in the if block are done
884  commandQueue_.push_front(poolstring_t(stringPool_, "DecreaseScopeLevel"));
885  // Add commands to commandQueue in reverse order,
886  // since pushing from the front
887  while (!commandBuffer_.is_empty()) {
888  commandQueue_.push_front(commandBuffer_.back());
889  commandBuffer_.pop_back();
890  }
891  }
892  else {
893  --currScopeLevel_;
894  }
895  exit_scope();
896  Log.trace(F("%s: closed\n"), PRINT_FUNC);
897  }

902  void create_else() {
903  enter_scope(InterpreterStatus::CREATING_ELSE);
904  Log.trace(F("%s\n"), PRINT_FUNC);
905  // Expand lastCondition_ if necessary
906  while (lastCondition_.size() <= currScopeLevel_ + 1) {

944  int condition = lastCondition_[currScopeLevel_ - 1];
945  // No else should be run after this
946  lastCondition_[currScopeLevel_ - 1] = -1;
947  // Only run if last condition was false
948  if (condition == 0) {
949  Log.trace(F("%s: will be executed\n"), PRINT_FUNC);
950  // Special interpreter-only command to ensure we decrease scope level

952  commandQueue_.push_front(poolstring_t(stringPool_, "DecreaseScopeLevel"));
953  // Add commands to commandQueue in reverse order,

958  }

960  else {

962  }
