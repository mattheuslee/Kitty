465  std::stack<Token> evaluate_postfix(std::queue<Token> & tokenQueue) {
466  std::stack<Token> tokenStack;
467 
468  while (!tokenQueue.empty()) {
469  Token token = tokenQueue.front();
470  tokenQueue.pop();
471  if (token.is_unary_operator()) {
472  Token operand = tokenStack.top();
473  tokenStack.pop();
474  Token result = evaluate_unary_operation(token, operand);
475  tokenStack.push(result);
476  }
477  else if (token.is_operator()) {
478  Token rhs = tokenStack.top();
479  tokenStack.pop();
480  Token lhs = tokenStack.top();
481  tokenStack.pop();
482  Token result = evaluate_operation(token, lhs, rhs);
483  //Serial.print(lhs.str().c_str());
484  //Serial.print(token.str().c_str());
485  //Serial.print(rhs.str().c_str());
486  //Serial.print(" = ");
487  //Serial.println(result.str().c_str());
488  tokenStack.push(result);
489  }
490  else if (token.is_operand()) {
491  tokenStack.push(token);
492  }
493  else if (token.is_function()) {
494  tokenStack.push(token);
495  }
496  }
497  return tokenStack;
498  }

544  Token evaluate_operation(Token const & op, Token const & lhs, Token const & rhs) {
545  int lhsValue = get_token_value(lhs);
546  int rhsValue = get_token_value(rhs);
547  Token result(TokenType::NUM_VAL);
548  result.value = "0";
549 
550  if (op.type == TokenType::EQUALS) {
551  result.value = int_to_str(lhsValue == rhsValue);
552  }
553  else if (op.type == TokenType::L_EQUALS) {
554  result.value = int_to_str(lhsValue <= rhsValue);
555  }
556  else if (op.type == TokenType::G_EQUALS) {
557  result.value = int_to_str(lhsValue >= rhsValue);
558  }
559  else if (op.type == TokenType::LESS) {
560  result.value = int_to_str(lhsValue < rhsValue);
561  }
562  else if (op.type == TokenType::GREATER) {
563  result.value = int_to_str(lhsValue > rhsValue);
564  }
565  else if (op.type == TokenType::MATH_ADD) {
566  result.value = int_to_str(lhsValue + rhsValue);
567  }
568  else if (op.type == TokenType::MATH_SUB) {
569  result.value = int_to_str(lhsValue - rhsValue);
570  }
571  else if (op.type == TokenType::MATH_MUL) {
572  result.value = int_to_str(lhsValue * rhsValue);
573  }
574  else if (op.type == TokenType::MATH_DIV) {
575  result.value = int_to_str(lhsValue / rhsValue);
576  }
577  else if (op.type == TokenType::MATH_MOD) {
578  result.value = int_to_str(lhsValue % rhsValue);
579  }
580  else if (op.type == TokenType::MATH_POW) {
581  result.value = int_to_str(power(lhsValue, rhsValue));
582  }
583  else if (op.type == TokenType::LOGI_AND) {
584  result.value = int_to_str(lhsValue && rhsValue);
585  }
586  else if (op.type == TokenType::LOGI_OR) {
587  result.value = int_to_str(lhsValue || rhsValue);
588  }
589  return result;
590  }
