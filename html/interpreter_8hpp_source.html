527  std::stack<Token> evaluate_postfix(std::queue<Token> & tokenQueue) {
528  std::stack<Token> tokenStack;
529 
530  while (!tokenQueue.empty()) {
531  Token token = tokenQueue.front();
532  tokenQueue.pop();
533  if (token.is_unary_operator()) {
534  Token operand = tokenStack.top();
535  tokenStack.pop();
536  Token result = evaluate_unary_operation(token, operand);
537  tokenStack.push(result);
538  }
539  else if (token.is_operator()) {
540  Token rhs = tokenStack.top();
541  tokenStack.pop();
542  Token lhs = tokenStack.top();
543  tokenStack.pop();
544  Token result = evaluate_operation(token, lhs, rhs);
545  tokenStack.push(result);
546  }
547  else if (token.is_operand()) {
548  tokenStack.push(token);
549  }
550  else if (token.is_function()) {
551  tokenStack.push(token);
552  }
553  }
554  return tokenStack;
555  }

601  Token evaluate_operation(Token const & op, Token const & lhs, Token const & rhs) {
602  int lhsValue = get_token_value(lhs);
603  int rhsValue = get_token_value(rhs);
604  Token result(TokenType::NUM_VAL);
605  result.value = "0";
606 
607  if (op.type == TokenType::EQUALS) {
608  result.value = int_to_str(lhsValue == rhsValue);
609  }
610  else if (op.type == TokenType::L_EQUALS) {
611  result.value = int_to_str(lhsValue <= rhsValue);
612  }
613  else if (op.type == TokenType::G_EQUALS) {
614  result.value = int_to_str(lhsValue >= rhsValue);
615  }
616  else if (op.type == TokenType::LESS) {
617  result.value = int_to_str(lhsValue < rhsValue);
618  }
619  else if (op.type == TokenType::GREATER) {
620  result.value = int_to_str(lhsValue > rhsValue);
621  }
622  else if (op.type == TokenType::MATH_ADD) {
623  result.value = int_to_str(lhsValue + rhsValue);
624  }
625  else if (op.type == TokenType::MATH_SUB) {
626  result.value = int_to_str(lhsValue - rhsValue);
627  }
628  else if (op.type == TokenType::MATH_MUL) {
629  result.value = int_to_str(lhsValue * rhsValue);
630  }
631  else if (op.type == TokenType::MATH_DIV) {
632  result.value = int_to_str(lhsValue / rhsValue);
633  }
634  else if (op.type == TokenType::MATH_MOD) {
635  result.value = int_to_str(lhsValue % rhsValue);
636  }
637  else if (op.type == TokenType::MATH_POW) {
638  result.value = int_to_str(power(lhsValue, rhsValue));
639  }
640  else if (op.type == TokenType::LOGI_AND) {
641  result.value = int_to_str(lhsValue && rhsValue);
642  }
643  else if (op.type == TokenType::LOGI_OR) {
644  result.value = int_to_str(lhsValue || rhsValue);
645  }
646  return result;
647  }
