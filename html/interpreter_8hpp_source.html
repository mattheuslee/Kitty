277  Log.verbose(F("%s\n"), PRINT_FUNC);

420  void execute_move_by(Deque<Token> const & command) {
421  Log.verbose(F("%s\n"), PRINT_FUNC);
422  Deque<Token> tokenQueue(command);
423  Token moveByToken = tokenQueue.back();
424  tokenQueue.pop_back();
425  PoolString name(tokenQueue.front().get_value());
426  tokenQueue.pop_front();
427  // Nothing to move
428  if (!number_exists(name) && !device_exists(name)) {
429  Serial.print(F("Error: "));
430  Serial.print(name.c_str());
431  Serial.println(F(" does not exist"));
432  return;
433  }
434 
435  Deque<Token> result = evaluate_postfix(tokenQueue);
436 
437  int displacement, durationMs = 0;
438  // There will be an additional argument if it is move by for
439  if (moveByToken.is_move_by_for()) {
440  durationMs = get_token_value(result.back());
441  result.pop_back();
442  }
443  displacement = get_token_value(result.back());
444 
445  // Execute move
446  int value = get_number_value(name);
447  int deviceInfo1 = get_device_info(name, 1);
448  int deviceInfo2 = get_device_info(name, 2);
449 
450  if (number_exists(name)) {
451  machineState_.set_number(name, value + displacement);
452  }
453  else {
454  switch (get_device_type(name)) {
455  case LED:
456  int brightness = deviceInfo2 + displacement;
457  if (brightness > 100) {
458  Log.notice(F("%s: LED brightness over 100%, maximum is 100%\n"), PRINT_FUNC);
459  brightness = 100;
460  }
461  else if (brightness < 0) {
462  Log.notice(F("%s: LED brightness below 0%, minimum is 0%\n"), PRINT_FUNC);
463  brightness = 0;
464  }
465  analogWrite(deviceInfo1, brightness * 2.55);
466  machineState_.set_device(name, DeviceType::LED, -1, deviceInfo1, brightness);
467  };
468  }
469 
470  if (command.back().is_move_by_for()) {
471  // Additional time delay
472  delay(durationMs);
473  // Then set back to original value
474  if (number_exists(name)) {
475  machineState_.set_number(name, value);
476  }
477  else {
478  switch (get_device_type(name)) {
479  case LED:
480  analogWrite(deviceInfo1, deviceInfo2 * 2.55);
481  machineState_.set_device(name, DeviceType::LED, -1, deviceInfo1, deviceInfo2);
482  };
483  }
484  }

495  void execute_run_group(Deque<Token> const & command) {
496  Log.verbose(F("%s\n"), PRINT_FUNC);
497  Deque<Token> tokenQueue(command);
498  // Remove RunGroup command from back
499  tokenQueue.pop_back();
500  // Extract name of group and check if it exists
501  PoolString name(tokenQueue.front().get_value());
502  if (!group_exists(name)) {
503  Log.warning(F("%s: %s does not exist\n"), PRINT_FUNC, name.c_str());
504  return;
505  }
506  // Extract number of times to run group
507  Deque<Token> result = evaluate_postfix(tokenQueue);
508  int numTimes = get_token_value(result.back());
509  // Push command for one more call to run the group
510  if (numTimes > 1) {
511  commandQueue_.push_front(PoolString(*getPoolFunc_));
512  commandQueue_.front() += name.c_str();
513  commandQueue_.front() += "RunGroup(";
514  commandQueue_.front() += int_to_str(numTimes - 1, *getPoolFunc_);
515  commandQueue_.front() += ")";
516  }
517  // Continuously run the group
518  else if (numTimes == -1) {
519  commandQueue_.push_front(PoolString(*getPoolFunc_));
520  commandQueue_.front() += name.c_str();
521  commandQueue_.front() += "RunGroup(-1)";
522  }
523  // If running group at least once(or continuously)
524  if (numTimes == -1 || numTimes > 0) {
525  Deque<PoolString> groupCommands = get_group_commands(name);
526  // Push from the last command to ensure correct order
527  typename Deque<PoolString>::Iterator it = groupCommands.end();
528  --it;
529  for ( ; it != groupCommands.begin(); --it) {
530  commandQueue_.push_front(*it);
531  }
532  // Additional push for the first command (not handled by loop)
533  commandQueue_.push_front(*it);
534  }
535  }

556  Deque<Token> evaluate_postfix(Deque<Token> const & tokenQueue) {
557  Log.verbose(F("%s\n"), PRINT_FUNC);
558  Deque<Token> tokenStack;
559 
560  for (typename Deque<Token>::ConstIterator it = tokenQueue.begin(); it != tokenQueue.end(); ++it) {
561  Token const & token = *it;
562  if (token.is_unary_operator()) {
563  Token operand = tokenStack.back();
564  tokenStack.pop_back();
565  Token result = evaluate_unary_operation(token, operand);
566  tokenStack.push_back(result);
567  }
568  else if (token.is_operator()) {
569  Token rhs = tokenStack.back();
570  tokenStack.pop_back();
571  Token lhs = tokenStack.back();
572  tokenStack.pop_back();
573  Token result = evaluate_operation(token, lhs, rhs);
574  tokenStack.push_back(result);
575  }
576  else if (token.is_operand()) {
577  // Instantly evaluate
578  tokenStack.push_back(Token(TokenType::NUM_VAL, int_to_str(get_token_value(token))));
579  }
580  // Do nothing if function, shouldn't have more than one function
581  // within a command, and the function should have already been
582  // extracted before calling evaluate_postfix
583  //else if (token.is_function()) {
584  // tokenStack.push_back(token);
585  //}
586  }
587  return tokenStack;
588  }

635  Token evaluate_operation(Token const & operation, Token const & lhs, Token const & rhs) {
636  Log.verbose(F("%s\n"), PRINT_FUNC);
637  int lhsValue = get_token_value(lhs);
638  int rhsValue = get_token_value(rhs);
639  Token result(TokenType::NUM_VAL);
640  result.set_value(int_to_str(0));
641 
642  if (operation.is_equals()) {
643  result.set_value(int_to_str(lhsValue == rhsValue));
644  }
645  else if (operation.is_l_equals()) {
646  result.set_value(int_to_str(lhsValue <= rhsValue));
647  }
648  else if (operation.is_g_equals()) {
649  result.set_value(int_to_str(lhsValue >= rhsValue));
650  }
651  else if (operation.is_less()) {
652  result.set_value(int_to_str(lhsValue < rhsValue));
653  }
654  else if (operation.is_greater()) {
655  result.set_value(int_to_str(lhsValue > rhsValue));
656  }
657  else if (operation.is_math_add()) {
658  result.set_value(int_to_str(lhsValue + rhsValue));
659  }
660  else if (operation.is_math_sub()) {
661  result.set_value(int_to_str(lhsValue - rhsValue));
662  }
663  else if (operation.is_math_mul()) {
664  result.set_value(int_to_str(lhsValue * rhsValue));
665  }
666  else if (operation.is_math_div()) {
667  result.set_value(int_to_str(lhsValue / rhsValue));
668  }
669  else if (operation.is_math_mod()) {
670  result.set_value(int_to_str(lhsValue % rhsValue));
671  }
672  else if (operation.is_math_pow()) {
673  result.set_value(int_to_str(power(lhsValue, rhsValue)));
674  }
675  else if (operation.is_logi_and()) {
676  result.set_value(int_to_str(lhsValue && rhsValue));
677  }
678  else if (operation.is_logi_or()) {
679  result.set_value(int_to_str(lhsValue || rhsValue));
680  }
681  return result;
682  }
