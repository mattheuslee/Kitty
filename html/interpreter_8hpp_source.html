272  void execute_command_tokens(Deque<Token> const & command) {
273  Log.verbose(F("%s\n"), PRINT_FUNC);
274  if (command.back().is_if()) {
275  execute_if(command);
276  }
277  else if (command.back().is_else()) {
278  execute_else(command);
279  }
280  else {
281  // For every other command, last condition at this scope level becomes null
282  lastCondition_[currScopeLevel_] = -1;
283  if (command.back().is_print()) {
284  execute_print(command);
285  }
286  else if (command.back().is_wait()) {
287  execute_wait(command);
288  }
289  else if (command.front().is_name()) {
290  // Printing information
291  if (command.size() == 1) {
292  execute_print_info(command);
293  }
294  // Creation of a device/group
295  else if (command.back().is_create_num() ||
296  command.back().is_create_led() ||
297  command.back().is_create_group()) {
298  execute_create(command);
299  }
300  else if (command.back().is_move_by() ||
301  command.back().is_move_by_for()) {
302  execute_move_by(command);
303  }
304  else if (command.back().is_set_to() ||
305  command.back().is_set_to_for()) {
306  execute_set_to(command);
307  }
308  // Running group
309  else {
310  execute_run_group(command);
311  }
312  }
313  else if (command.front().is_string()) {
314  execute_print_string(command);
315  }

461  tokenQueue.pop_back();

473  Deque<Token> result = evaluate_postfix(tokenQueue);
474  int displacement, durationMs = 0;

477  durationMs = get_token_value(result.back());
478  result.pop_back();
479  }

495  brightness = 100;
496  }
497  else if (brightness < 0) {
498  Log.notice(F("%s: LED brightness below 0%, minimum is 0%\n"), PRINT_FUNC);
499  brightness = 0;
500  }

502  machineState_.set_device(name, DeviceType::LED, -1, deviceInfo1, brightness);
503  };
504  }
505  // MoveByFor command
506  if (moveByToken.is_move_by_for()) {
507  // Additional time delay
508  delay(durationMs);
509  // Then set back to original value
510  if (number_exists(name)) {
511  machineState_.set_number(name, value);
512  }
513  else {
514  switch (get_device_type(name)) {

602  void execute_run_group(Deque<Token> const & command) {
603  Log.verbose(F("%s\n"), PRINT_FUNC);
604  Deque<Token> tokenQueue(command);
605  // Remove RunGroup command from back
606  tokenQueue.pop_back();
607  // Extract name of group and check if it exists
608  PoolString name(tokenQueue.front().get_value());
609  if (!group_exists(name)) {
610  Log.warning(F("%s: %s does not exist\n"), PRINT_FUNC, name.c_str());
611  return;
612  }
613  // Extract number of times to run group
614  Deque<Token> result = evaluate_postfix(tokenQueue);
615  int numTimes = get_token_value(result.back());
616  // Push command for one more call to run the group
617  if (numTimes > 1) {
618  commandQueue_.push_front(PoolString(*getPoolFunc_));
619  commandQueue_.front() += name.c_str();
620  commandQueue_.front() += "RunGroup(";
621  commandQueue_.front() += int_to_str(numTimes - 1, *getPoolFunc_);
622  commandQueue_.front() += ")";
623  }
624  // Continuously run the group
625  else if (numTimes == -1) {
626  commandQueue_.push_front(PoolString(*getPoolFunc_));
627  commandQueue_.front() += name.c_str();
628  commandQueue_.front() += "RunGroup(-1)";
629  }
630  // If running group at least once(or continuously)
631  if (numTimes == -1 || numTimes > 0) {
632  Deque<PoolString> groupCommands = get_group_commands(name);
633  // Push from the last command to ensure correct order
634  typename Deque<PoolString>::Iterator it = groupCommands.end();
635  --it;
636  for ( ; it != groupCommands.begin(); --it) {
637  commandQueue_.push_front(*it);
638  }
639  // Additional push for the first command (not handled by loop)
640  commandQueue_.push_front(*it);
641  }
642  }

663  Deque<Token> evaluate_postfix(Deque<Token> const & tokenQueue) {
664  Log.verbose(F("%s\n"), PRINT_FUNC);
665  Deque<Token> tokenStack;
666 
667  for (typename Deque<Token>::ConstIterator it = tokenQueue.begin(); it != tokenQueue.end(); ++it) {
668  Token const & token = *it;
669  if (token.is_unary_operator()) {
670  Token operand = tokenStack.back();
671  tokenStack.pop_back();
672  Token result = evaluate_unary_operation(token, operand);
673  tokenStack.push_back(result);
674  }
675  else if (token.is_operator()) {
676  Token rhs = tokenStack.back();
677  tokenStack.pop_back();
678  Token lhs = tokenStack.back();
679  tokenStack.pop_back();
680  Token result = evaluate_operation(token, lhs, rhs);
681  tokenStack.push_back(result);
682  }
683  else if (token.is_operand()) {
684  // Instantly evaluate
685  tokenStack.push_back(Token(TokenType::NUM_VAL, int_to_str(get_token_value(token))));
686  }
687  // Everything else just goes directly to the tokenStack
688  else {
689  tokenStack.push_back(token);
690  }
691  }
692  return tokenStack;
693  }

708  Token evaluate_unary_operation(Token const & operation, Token const & operand) {
709  Log.verbose(F("%s\n"), PRINT_FUNC);

718  result.set_value(int_to_str(!value));

740  Token evaluate_operation(Token const & operation, Token const & lhs, Token const & rhs) {
741  Log.verbose(F("%s\n"), PRINT_FUNC);
742  int lhsValue = get_token_value(lhs);
743  int rhsValue = get_token_value(rhs);
744  Token result(TokenType::NUM_VAL);
745  result.set_value(int_to_str(0));
746 
747  if (operation.is_equals()) {
748  result.set_value(int_to_str(lhsValue == rhsValue));
749  }
750  else if (operation.is_l_equals()) {
751  result.set_value(int_to_str(lhsValue <= rhsValue));
752  }
753  else if (operation.is_g_equals()) {
754  result.set_value(int_to_str(lhsValue >= rhsValue));
755  }
756  else if (operation.is_less()) {
757  result.set_value(int_to_str(lhsValue < rhsValue));
758  }
759  else if (operation.is_greater()) {
760  result.set_value(int_to_str(lhsValue > rhsValue));
761  }
762  else if (operation.is_math_add()) {
763  result.set_value(int_to_str(lhsValue + rhsValue));
764  }
765  else if (operation.is_math_sub()) {
766  result.set_value(int_to_str(lhsValue - rhsValue));
767  }
768  else if (operation.is_math_mul()) {
769  result.set_value(int_to_str(lhsValue * rhsValue));
770  }
771  else if (operation.is_math_div()) {
772  result.set_value(int_to_str(lhsValue / rhsValue));
773  }
774  else if (operation.is_math_mod()) {
775  result.set_value(int_to_str(lhsValue % rhsValue));
776  }
777  else if (operation.is_math_pow()) {
778  result.set_value(int_to_str(power(lhsValue, rhsValue)));
779  }
780  else if (operation.is_logi_and()) {
781  result.set_value(int_to_str(lhsValue && rhsValue));
782  }
783  else if (operation.is_logi_or()) {
784  result.set_value(int_to_str(lhsValue || rhsValue));
785  }
786  return result;
787  }

896  void add_to_if(PoolString const & command) {
897  Log.verbose(F("%s\n"), PRINT_FUNC);
898  Deque<Token> tokens = tokenizer_.tokenize(command);
899  // Group is closed
900  if (bracketParity_ == 0 && tokens.size() == 2 && tokens.front().is_cl_paren()) {
901  close_if();
902  }
903  // Add command to group
904  else {
905  commandBuffer_.push_back(command);
906  // Check if command introduces a '('
907  if (tokens[tokens.size() - 2].is_op_paren()) {
908  ++bracketParity_;
909  }
910  // Check if command is a single ')'
911  else if (tokens.size() == 2 && tokens.front().is_cl_paren()) {
912  --bracketParity_;
913  }
914  }
915  }

921  void close_if() {
922  Log.verbose(F("%s\n"), PRINT_FUNC);
923  int condition = lastCondition_[currScopeLevel_ - 1];
924  if (condition) {
925  // Special interpreter-only command to ensure we decrease scope level
926  // after all the instructions in the if block are done
927  commandQueue_.push_front(PoolString("DecreaseScopeLevel", *getPoolFunc_));
928  // Add commands to commandQueue in reverse order,
929  // since pushing from the front
930  while (!commandBuffer_.is_empty()) {
931  commandQueue_.push_front(commandBuffer_.back());
932  commandBuffer_.pop_back();
933  }
934  }
935  else {
936  --currScopeLevel_;

961  Log.verbose(F("%s\n"), PRINT_FUNC);

963  // Group is closed
964  if (bracketParity_ == 0 && tokens.size() == 2 && tokens.front().is_cl_paren()) {
965  close_else();
966  }

968  else {

972  ++bracketParity_;
973  }

987  Log.verbose(F("%s\n"), PRINT_FUNC);
988  int condition = lastCondition_[currScopeLevel_ - 1];
989  // No else should be run after this
990  lastCondition_[currScopeLevel_ - 1] = -1;
991  // Only run if last condition was false
992  if (condition == 0) {
993  // Special interpreter-only command to ensure we decrease scope level
994  // after all the instructions in the if block are done
995  commandQueue_.push_front(PoolString("DecreaseScopeLevel", *getPoolFunc_));
996  // Add commands to commandQueue in reverse order,
997  // since pushing from the front
998  while (!commandBuffer_.is_empty()) {
999  commandQueue_.push_front(commandBuffer_.back());
1000  commandBuffer_.pop_back();
1001  }
1002  }
1003  else {

1005  }
1006  exit_scope();
1007  }
