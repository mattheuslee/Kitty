388  Token evaluate_operation(Token const & op, Token const & lhs, Token const & rhs) {

390  int rhsValue = get_token_value(rhs);
391  Token result(TokenType::NUM_VAL);
392  result.value = "0";

394  if (op.type == TokenType::EQUALS) {
395  result.value = int_to_str(lhsValue == rhsValue);
396  }
397  else if (op.type == TokenType::L_EQUALS) {
398  result.value = int_to_str(lhsValue <= rhsValue);
399  }
400  else if (op.type == TokenType::G_EQUALS) {
401  result.value = int_to_str(lhsValue >= rhsValue);
402  }
403  else if (op.type == TokenType::LESS) {
404  result.value = int_to_str(lhsValue < rhsValue);
405  }
406  else if (op.type == TokenType::GREATER) {
407  result.value = int_to_str(lhsValue > rhsValue);
408  }
409  else if (op.type == TokenType::MATH_ADD) {
410  result.value = int_to_str(lhsValue + rhsValue);
411  }
412  else if (op.type == TokenType::MATH_SUB) {
413  result.value = int_to_str(lhsValue - rhsValue);
414  }
415  else if (op.type == TokenType::MATH_MUL) {
416  result.value = int_to_str(lhsValue * rhsValue);
417  }
418  else if (op.type == TokenType::MATH_DIV) {
419  result.value = int_to_str(lhsValue / rhsValue);
420  }
421  else if (op.type == TokenType::MATH_MOD) {
422  result.value = int_to_str(lhsValue % rhsValue);
