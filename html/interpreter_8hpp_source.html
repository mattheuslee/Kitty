540  std::stack<Token> evaluate_postfix(std::queue<Token> & tokenQueue) {
541  std::stack<Token> tokenStack;
542 
543  while (!tokenQueue.empty()) {
544  Token token = tokenQueue.front();
545  tokenQueue.pop();
546  if (token.is_unary_operator()) {
547  Token operand = tokenStack.top();
548  tokenStack.pop();
549  Token result = evaluate_unary_operation(token, operand);
550  tokenStack.push(result);
551  }
552  else if (token.is_operator()) {
553  Token rhs = tokenStack.top();
554  tokenStack.pop();
555  Token lhs = tokenStack.top();
556  tokenStack.pop();
557  Token result = evaluate_operation(token, lhs, rhs);
558  tokenStack.push(result);
559  }
560  else if (token.is_operand()) {
561  tokenStack.push(token);
562  }
563  else if (token.is_function()) {
564  tokenStack.push(token);
565  }
566  }
567  return tokenStack;
568  }

614  Token evaluate_operation(Token const & op, Token const & lhs, Token const & rhs) {
615  int lhsValue = get_token_value(lhs);
616  int rhsValue = get_token_value(rhs);
617  Token result(TokenType::NUM_VAL);
618  result.value = "0";
619 
620  if (op.type == TokenType::EQUALS) {
621  result.value = int_to_str(lhsValue == rhsValue);
622  }
623  else if (op.type == TokenType::L_EQUALS) {
624  result.value = int_to_str(lhsValue <= rhsValue);
625  }
626  else if (op.type == TokenType::G_EQUALS) {
627  result.value = int_to_str(lhsValue >= rhsValue);
628  }
629  else if (op.type == TokenType::LESS) {
630  result.value = int_to_str(lhsValue < rhsValue);
631  }
632  else if (op.type == TokenType::GREATER) {
633  result.value = int_to_str(lhsValue > rhsValue);
634  }
635  else if (op.type == TokenType::MATH_ADD) {
636  result.value = int_to_str(lhsValue + rhsValue);
637  }
638  else if (op.type == TokenType::MATH_SUB) {
639  result.value = int_to_str(lhsValue - rhsValue);
640  }
641  else if (op.type == TokenType::MATH_MUL) {
642  result.value = int_to_str(lhsValue * rhsValue);
643  }
644  else if (op.type == TokenType::MATH_DIV) {
645  result.value = int_to_str(lhsValue / rhsValue);
646  }
647  else if (op.type == TokenType::MATH_MOD) {
648  result.value = int_to_str(lhsValue % rhsValue);
649  }
650  else if (op.type == TokenType::MATH_POW) {
651  result.value = int_to_str(power(lhsValue, rhsValue));
652  }
653  else if (op.type == TokenType::LOGI_AND) {
654  result.value = int_to_str(lhsValue && rhsValue);
655  }
656  else if (op.type == TokenType::LOGI_OR) {
657  result.value = int_to_str(lhsValue || rhsValue);
658  }
659  return result;
660  }
