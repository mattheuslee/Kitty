471  void execute_run_group(std::vector<Token> const & command) {
472  std::queue<Token> tokenQueue;
473  for (int i = 0; i < command.size() - 1; ++i) {
474  tokenQueue.push(command[i]);
475  }
476  // Extract name of group and check if it exists
477  poolstring_t name(stringPool_, tokenQueue.front().value.c_str());
478  int groupIdx = group_exists(name);
479  if (groupIdx == -1) {
480  Log.warning(F("Interpreter::execute_run_group %s does not exist\n"), name.c_str());
481  return;
482  }
483  // Extract number of times to run group
484  std::stack<Token> result = evaluate_postfix(tokenQueue);
485  int numTimes = str_to_int(result.top().value);
486  Log.trace(F("Running group = %s, %d times\n"), name.c_str(), numTimes);
487  // Push command for one more call to run the group
488  if (numTimes > 1) {
489  poolstring_t repeatGroupCommand(stringPool_);
490  repeatGroupCommand += name.c_str();
491  repeatGroupCommand += "RunGroup(";
492  repeatGroupCommand += int_to_str(numTimes - 1, stringPool_);
493  repeatGroupCommand += ")";
494  commandQueue_.push_front(repeatGroupCommand);
495  Log.trace(F("Interpreter::execute_run_group pushing %s to the front of the command queue for next round\n"), repeatGroupCommand.c_str());
496  }
497  // Push commands from group to command queue
498  int numGroupCommands = groupCommands_.size(groupIdx);
499  Log.trace(F("Interpreter::execute_run_group %d commands to add to command queue\n"), numGroupCommands);
500  for (int i = numGroupCommands - 1; i >= 0; --i) {
501  poolstring_t commandString = groupCommands_.get_str(groupIdx, i);
502  Log.trace(F("Interpreter::execute_run_group pushing %s to front of command queue\n"), commandString.c_str());
503  commandQueue_.push_front(commandString);
504  }
505  }

526  std::stack<Token> evaluate_postfix(std::queue<Token> & tokenQueue) {
527  std::stack<Token> tokenStack;
528 
529  while (!tokenQueue.empty()) {
530  Token token = tokenQueue.front();
531  tokenQueue.pop();
532  if (token.is_unary_operator()) {
533  Token operand = tokenStack.top();
534  tokenStack.pop();
535  Token result = evaluate_unary_operation(token, operand);
536  tokenStack.push(result);
537  }
538  else if (token.is_operator()) {
539  Token rhs = tokenStack.top();
540  tokenStack.pop();
541  Token lhs = tokenStack.top();
542  tokenStack.pop();
543  Token result = evaluate_operation(token, lhs, rhs);
544  tokenStack.push(result);
545  }
546  else if (token.is_operand()) {
547  tokenStack.push(token);
548  }
549  else if (token.is_function()) {
550  tokenStack.push(token);
551  }
552  }
553  return tokenStack;
554  }

600  Token evaluate_operation(Token const & op, Token const & lhs, Token const & rhs) {
601  int lhsValue = get_token_value(lhs);
602  int rhsValue = get_token_value(rhs);
603  Token result(TokenType::NUM_VAL);
604  result.value = "0";
605 
606  if (op.type == TokenType::EQUALS) {
607  result.value = int_to_str(lhsValue == rhsValue);
608  }
609  else if (op.type == TokenType::L_EQUALS) {
610  result.value = int_to_str(lhsValue <= rhsValue);
611  }
612  else if (op.type == TokenType::G_EQUALS) {
613  result.value = int_to_str(lhsValue >= rhsValue);
614  }
615  else if (op.type == TokenType::LESS) {
616  result.value = int_to_str(lhsValue < rhsValue);
617  }
618  else if (op.type == TokenType::GREATER) {
619  result.value = int_to_str(lhsValue > rhsValue);
620  }
621  else if (op.type == TokenType::MATH_ADD) {
622  result.value = int_to_str(lhsValue + rhsValue);
623  }
624  else if (op.type == TokenType::MATH_SUB) {
625  result.value = int_to_str(lhsValue - rhsValue);
626  }
627  else if (op.type == TokenType::MATH_MUL) {
628  result.value = int_to_str(lhsValue * rhsValue);
629  }
630  else if (op.type == TokenType::MATH_DIV) {
631  result.value = int_to_str(lhsValue / rhsValue);
632  }
633  else if (op.type == TokenType::MATH_MOD) {
634  result.value = int_to_str(lhsValue % rhsValue);
635  }
636  else if (op.type == TokenType::MATH_POW) {
637  result.value = int_to_str(power(lhsValue, rhsValue));
638  }
639  else if (op.type == TokenType::LOGI_AND) {
640  result.value = int_to_str(lhsValue && rhsValue);
641  }
642  else if (op.type == TokenType::LOGI_OR) {
643  result.value = int_to_str(lhsValue || rhsValue);
644  }
645  return result;
646  }

926  Log.trace(F("Interpreter::add_to_group(%s)\n"), command.c_str());
927  std::vector<Token> tokens = tokenizer_.tokenize(std::string(command.c_str()));
928  // Group is closed
929  if (bracketParity_ == 0 && tokens.size() == 2 && tokens[0].is_right_bracket()) {
930  Log.trace(F("Interpreter::add_to_group group will be closed\n"));
931  close_group();
932  }
933  // Add command to group
934  else {
935  Log.trace(F("Interpreter::add_to_group adding %s to group %s\n"), command.c_str(), lastGroupName_.c_str());
936  int groupIdx = group_exists(lastGroupName_);
937  groupCommands_.push_back(groupIdx, command);
938  // Check if command introduces a '('
939  if (tokens[tokens.size() - 2].is_left_bracket()) {
940  ++bracketParity_;
941  }
942  // Check if command closes with a ')'
943  else if (tokens.size() == 2 && tokens[0].is_right_bracket()) {
944  --bracketParity_;
945  }
946  Log.trace(F("Interpreter::add_to_group bracket parity %d\n"), bracketParity_);
947  }
948  }
